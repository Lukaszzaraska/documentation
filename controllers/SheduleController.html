import { Body, Controller, Get, Param, Patch, Post, Req, UseFilters, UseGuards } from '@nestjs/common';
import { RolesGuard } from 'src/Auth/roles.guard';
import { SheduleService } from './shedule.service';
import { Request } from 'express';
import { Role } from '@prisma/client';
import { Roles } from 'src/Auth/decorators/roles.decorator';
import { HttpExceptionFilter } from 'src/common/validation/http-exception.filter';
import { CreateSheduleDto } from './dto/createShedule.dto';
import { EditSheduleDto } from './dto/editShedule.dto';

@Controller('shedule')
@UseGuards(RolesGuard)
export class SheduleController {
    constructor(private sheduleService: SheduleService) { }

    //#TODO Zrobić jeszcze dla mgtmt ale narazie nie wiadomo jak ma się to wyświetlać :)

    @UseFilters(new HttpExceptionFilter())
    @Roles(Role.RD, Role.CA)
    @Get('findall/:IdSchool')
    async findAll(@Req() req: Request, @Param('IdSchool') IdSchool: string) {
        const region: string = req['user'].region;
        return this.sheduleService.findAll(region,IdSchool);
    }
    @UseFilters(new HttpExceptionFilter())
    @Roles(Role.RD, Role.CA)
    @Get('find/:IdShedule')
    async findById(@Req() req: Request, @Param('IdShedule') IdShedule: string) {
        const region: string = req['user'].region;
        return this.sheduleService.findById(region, IdShedule)
    }

    @UseFilters(new HttpExceptionFilter())
    @Roles(Role.RD, Role.CA)
    @Patch('edit/:IdShedule')
    async edit (@Body() EditSheduleDto: EditSheduleDto, @Param('IdShedule') IdShedule: string) {

        return this.sheduleService.edit(EditSheduleDto, IdShedule)
    }

    @UseFilters(new HttpExceptionFilter())
    @Roles(Role.RD, Role.CA)
    @Post('create/:IdSchool')
    async create(@Body() createSheduleDto: CreateSheduleDto, @Req() req: Request, @Param('IdSchool') IdSchool: string) {
        const region: string = req['user'].region;
        return this.sheduleService.create(createSheduleDto, IdSchool, region)
    }


    //wprowadzanie danych przez dr - dany region
    //wyświetlanie wszystkich planów zajęc z danej szkoły - i region
    // modyfikacja danego planu zajęc
}
