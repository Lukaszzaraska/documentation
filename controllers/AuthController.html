import {
    Body,
    Controller,
    Get,
    HttpCode,
    HttpStatus,
    Post,
    UseGuards,
    UseFilters,
    ForbiddenException,
    Delete,
    UsePipes,
    ValidationPipe
} from '@nestjs/common';
import { Public } from './decorators/public.decorator';
import { AuthService } from './auth.service';
import { AuthGuard } from './auth.guard';
import { LoginUserDto } from 'src/Auth/dto/authUser.dto';
import { HttpExceptionFilter } from 'src/common/validation/http-exception.filter';
import { GetUserDto } from 'src/models/user/dto/GetUser.dto';
import { Role } from '@prisma/client';
import { Roles } from './decorators/roles.decorator';
import { RolesGuard } from './roles.guard';
import { ChangePasswordDto } from './dto/changePassword.dto';
import { ReqInfo, ReqUserInfo } from './decorators/reqinfo.decorator';
import { AccessTokenDto } from './dto/authRefreshToken.dto';

@UseGuards(RolesGuard)
@Controller('auth')
export class AuthController {
    constructor(private authService: AuthService) { }

    @Public()
    @HttpCode(HttpStatus.OK)
    @UseFilters(new HttpExceptionFilter())
    @Post('login')
    async signIn(@Body() signInDto: LoginUserDto): Promise<AccessTokenDto> {
        return this.authService.signIn(signInDto.email, signInDto.password);
    }

    @UseFilters(new HttpExceptionFilter())
    @Roles(Role.RD, Role.CA, Role.MGTMT, Role.USER)
    @UsePipes(new ValidationPipe({ whitelist: true }))
    @Post('changePassword')
    async ChangePassword(@Body() Data: ChangePasswordDto, @ReqInfo() reqInfo: ReqUserInfo):Promise<GetUserDto> {
        return this.authService.ChangePassword(Data, reqInfo.id);
    }

    @Public()
    @HttpCode(HttpStatus.OK)
    @UseFilters(new HttpExceptionFilter())
    @Post('refreshToken')
    refreshToken(@Body() OldToken: AccessTokenDto): Promise<AccessTokenDto> {
        return this.authService.refreshToken(OldToken);
    }

   
    @Roles(Role.RD, Role.CA, Role.MGTMT, Role.USER)
    @UseFilters(new HttpExceptionFilter())
    @Delete('logOut')
    LogOut(): Promise<void> {
        return this.authService.LogOut();
    }

    @UseGuards(AuthGuard)
    @UseFilters(new HttpExceptionFilter())
    @Get('profile')
    getProfile(@ReqInfo() reqInfo: ReqUserInfo): Promise<GetUserDto> {
        return this.authService.Profile(reqInfo.email);
    }
}
