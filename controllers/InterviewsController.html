import { Body, Controller, ForbiddenException, Get, Param, Patch, Post, Query, UseFilters, UseGuards } from '@nestjs/common';
import { InterviewsService } from './interviews.service';
import { RolesGuard } from 'src/Auth/roles.guard';
import { Role } from '@prisma/client';
import { Roles } from 'src/Auth/decorators/roles.decorator';
import { CreateInterViewDto } from './dto/createInterView.dto';
import { AddUserInterViewDto } from './dto/addUserInterView.dto';
import { EditInterViewDto } from './dto/editInterView.dto';
import { HttpExceptionFilter } from 'src/common/validation/http-exception.filter';
import { RemoveUserInterViewDto } from './dto/removeUserInterview.dto';
import { GetUserInterViewDto } from './dto/getUserInterView.dto';
import { ReqInfo, ReqUserInfo } from 'src/Auth/decorators/reqinfo.decorator';
import { GetInterViewsDto } from './dto/GetInterViews.dto';
import { OptionsPipe } from '../fiszka/options.pipe';
import { GetFindByIdDto } from './dto/getFindById.dto';

@UseGuards(RolesGuard)
@Controller('interviews')
export class InterviewsController {
    constructor(private interviewsService: InterviewsService) { }


    @Roles(Role.CA, Role.USER, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Get('findallbyuser')
    async findAllByUser(@ReqInfo() reqInfo: ReqUserInfo): Promise<GetUserInterViewDto[]> {
        return this.interviewsService.findAllByUser(reqInfo.id)
    }


    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Get('findallbyregion')
    async findAllByRegion(@ReqInfo() reqInfo: ReqUserInfo): Promise<GetInterViewsDto[]> {
        return this.interviewsService.findAllByRegion(reqInfo.region)
    }

    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Get('findbyid/:idinterview')
    async findById(@Param('idinterview') IdInterviews: string, @ReqInfo() reqInfo: ReqUserInfo): Promise<GetFindByIdDto> {
        return this.interviewsService.findById(reqInfo.region, IdInterviews)
    }


    @Roles(Role.RD, Role.CA, Role.USER)
    @UseFilters(new HttpExceptionFilter())
    @Get('findallbyschool/:idschool')
    async findMyBySchool(
        @Param('idschool') idSchool: string,
        @ReqInfo() reqInfo: ReqUserInfo,
        @Query('self', OptionsPipe) self: number
    ): Promise<GetUserInterViewDto[]> {
        if (reqInfo.role.match(Role.RD)) {
            return this.interviewsService.findAllBySchool(idSchool);
        }
        const userId = reqInfo.id;
        const result = self ? this.interviewsService.findMyBySchool(idSchool, userId) : this.interviewsService.findAllBySchool(idSchool);
        return result;

    }

    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Get('findallbyuser/:iduser')
    async findByUser(@Param('iduser') idUser: string): Promise<GetUserInterViewDto[]> {
        return this.interviewsService.findByUser(idUser)
    }

    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Patch('edit/:idinterview')
    async editInterview(@Param('idinterview') idInterView: string, @Body() editInterViewDto: EditInterViewDto): Promise<EditInterViewDto> {
        return this.interviewsService.editInterview(idInterView, editInterViewDto)
    }
    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Patch('adduser/:idinterview')
    async addUserToInterview(@Param('idinterview') idInterView: string, @Body() addUserInterViewDto: AddUserInterViewDto): Promise<ForbiddenException|GetUserInterViewDto > {
        return this.interviewsService.addUserToInterview(idInterView, addUserInterViewDto)
    }
    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Patch('removeuser/:idinterview')
    async removeUserToInterview(@Param('idinterview') idInterView: string, @Body() removeUserInterViewDto: RemoveUserInterViewDto): Promise<RemoveUserInterViewDto> {
        return this.interviewsService.removeUserFromInterview(idInterView, removeUserInterViewDto)
    }
    @Roles(Role.CA, Role.RD)
    @UseFilters(new HttpExceptionFilter())
    @Post('create')
    async createInterview(@Body() createInterViewDto: CreateInterViewDto) {
        return this.interviewsService.CreateInterview(createInterViewDto)
    }

}
