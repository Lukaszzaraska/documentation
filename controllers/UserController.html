import { BadRequestException, Body, Controller, Get, Param, Patch, Post, Req, UseFilters, UseGuards } from '@nestjs/common';
import { UserService } from './user.service';
import { CreateUserForMGTMTDto, CreateUserForRdDto } from './dto/createUser.dto';
import { Role } from '@prisma/client';
import { Request } from 'express';
import { RolesGuard } from './../../Auth/roles.guard';
import { EditUserForMGTMTDto, EditUserForRdDto } from './dto/editUser.dto';
import { GetUserDto } from './dto/GetUser.dto';
import { HttpExceptionFilter } from './../../common/validation/http-exception.filter';
import { Roles } from 'src/Auth/decorators/roles.decorator';
import { UpperCasePipe } from './upperCasePipe.pipe';
import { ClassValidator } from 'src/common/helpers/ClassValidator';
import { ValidationdDto } from './dto/validation.dto';


@Controller('user')
@UseGuards(RolesGuard)
export class UserController {
  constructor(private userService: UserService) { }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA, Role.MGTMT)
  @Post('validation')
  async validation(@Body() ValidationUserDto: ValidationdDto): Promise<{InDatabase: boolean}> {
    return this.userService.Validation(ValidationUserDto)
  }



  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA, Role.MGTMT)
  @Get('findall')
  async findAll(@Req() req: Request): Promise<GetUserDto[]> {
    const role: Role = req['user'].role
    if (role.match(Role.MGTMT)) {
      return this.userService.findAll();
    } else {
      const region: string = req['user'].region;
      return this.userService.findByRegion(region, role);
    }
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT)
  @Get('region/:region')
  async findAllRegion(@Param('region') Region: string): Promise<GetUserDto[]> {
    return this.userService.findByRegion(Region);
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA, Role.MGTMT)
  @Get(':id')
  async findById(@Param('id') Id: string, @Req() req: Request): Promise<GetUserDto> {
    const role: Role = req['user'].role
    if (role.match(Role.MGTMT)) {
      return this.userService.findById(Id);
    } else {
      const region: string = req['user'].region;
      return this.userService.findById(Id, region);
    }
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER, Role.MGTMT, Role.CA, Role.RD)
  @Get('total')
  async findTotalById(@Req() req: Request): Promise<number> {
    const UserId: string = req['user'].id
    return this.userService.findTotalById(UserId);
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.MGTMT)
  @Post()
  async createUser(@Body(UpperCasePipe) createUserDto: any, @Req() req: Request): Promise<GetUserDto> {
    const role: Role = req['user'].role
    if (role.match(Role.MGTMT)) {
      const validationResult = await ClassValidator(createUserDto, CreateUserForMGTMTDto);
      if (validationResult instanceof BadRequestException) {
        throw validationResult;
      }
      const createUser: CreateUserForMGTMTDto = createUserDto
      return this.userService.CreateUserForMGTMT(createUser);
    } else {
      const validationResult = await ClassValidator(createUserDto, CreateUserForRdDto);
      if (validationResult instanceof BadRequestException) {
        throw validationResult;
      }
      const createUser: CreateUserForRdDto = createUserDto
      const region: string = req['user'].region
      return this.userService.CreateUserForRd(createUser, region);
    }
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA, Role.MGTMT)
  @Patch(':id')
  async editUser(@Param('id') Id: string, @Body(UpperCasePipe) editUserDto: any, @Req() req: Request) {
    const role: Role = req['user'].role
    if (role.match(Role.MGTMT)) {
      const validationResult = await ClassValidator(editUserDto, EditUserForMGTMTDto);
      if (validationResult instanceof BadRequestException) {
        throw validationResult;
      }
      const editUser: EditUserForMGTMTDto = editUserDto
      return this.userService.EditUserForMGTMT(editUser, Id)
    } else {
      const validationResult = await ClassValidator(editUserDto, EditUserForRdDto);
      if (validationResult instanceof BadRequestException) {
        throw validationResult;
      }
      const editUser: EditUserForRdDto = editUserDto
      const region: string = req['user'].region;
      return this.userService.EditUserForRd(editUser, Id, region, role)
    }
  }

}
