import { Body, Controller, Get, Param, Patch, Post, Req, UseFilters, UseGuards } from '@nestjs/common';
import { RolesGuard } from 'src/Auth/roles.guard';
import { FiszkaService } from './fiszka.service';
import { Request } from 'express';
import { Role } from '@prisma/client';
import { Roles } from 'src/Auth/decorators/roles.decorator';
import { HttpExceptionFilter } from 'src/common/validation/http-exception.filter';
import { GetFiszkaDto } from './dto/GetFiszka.dto';
import { EditFiszkaDto } from './dto/EditFiszka.dto';
import { CreateFiszkaDto } from './dto/CreateFiszka.dto';
import { AddgroupDto } from './dto/addGroup.dto';


@Controller('fiszka')
@UseGuards(RolesGuard)
export class FiszkaController {
  constructor(private fiszkaService: FiszkaService) { }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.RD, Role.CA)
  @Get('findall/:idInterview/:idEmployee')
  async findAllByInterview(@Param('idInterview') idInterview: string, @Param('idEmployee') UserID: string): Promise<GetFiszkaDto[]> {
    return this.fiszkaService.findAllByInterview(idInterview, UserID)
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA, Role.USER)
  @Get('findall/:idInterview')
  async findAllByUser(@Param('idInterview') idInterview: string, @Req() req: Request): Promise<GetFiszkaDto[]> {
    const UserID: string = req['user'].id;
    return this.fiszkaService.findAllByUser(idInterview, UserID)
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.RD, Role.CA)
  @Get('findallforuser/:idEmployee')
  async findAllForUser(@Param('idEmployee') UserID: string): Promise<GetFiszkaDto[]> {
    return this.fiszkaService.findAllForUser(UserID)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER)
  @Get('findallbyme/:idInterview')
  async findAllByMe(@Param('idInterview') idInterview: string, @Req() req: Request): Promise<GetFiszkaDto[]> {
    const UserID: string = req['user'].id;
    return this.fiszkaService.findAllByMe(idInterview, UserID)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.CA, Role.RD)
  @Patch('editfiszka/:idFiszka')
  async editFiszkaRd(@Param('idFiszka') idFiszka: string, @Req() req: Request, @Body() body: EditFiszkaDto) {
    const UserID: string = req['user'].id;
    return this.fiszkaService.editFiszkaRd(idFiszka, UserID, body)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER)
  @Patch('editfiszka/:idFiszka/:idInterview')
  async editFiszkaUser(@Param('idFiszka') idFiszka: string, @Param('idInterview') idInterview: string, @Req() req: Request, @Body() body: EditFiszkaDto) {
    const UserID: string = req['user'].id;
    return this.fiszkaService.editFiszkaUser(idFiszka, UserID, idInterview, body)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER, Role.MGTMT, Role.CA, Role.RD)
  @Post('addfiszki/:idInterview')
  async addFiszki(@Param('idInterview') idInterview: string, @Req() req: Request, @Body() body: CreateFiszkaDto[]) {
    const UserID: string = req['user'].id;
    return this.fiszkaService.addFiszki(idInterview, UserID, body)
  }

  //do testu
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.CA, Role.RD)
  @Patch('addgroup/:idgroup')
  async addGroup(@Param('idgroup') IdGroup: string, @Body() body: AddgroupDto) {
    return this.fiszkaService.addGroup(IdGroup, body)
  }
  // dodanie fiszki do danej grupy

  //#TODO: Tworzenie fiszek PR, AC, DR  |id wywiadówki

  //#TODO: Lista fiszek z danej wywiadówki per użytkopwnik PR | Id wywiadówki
  //#TODO: Lista fiszek danego pracownika  AC, DR | Id pracownika 
  //#TODO: Lista swoich fiszek  PR | Id wywiadówki
}
