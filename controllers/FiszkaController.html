import { Body, Controller, ForbiddenException, Get, Param, Patch, Post, UseFilters, UseGuards, Delete, Query } from '@nestjs/common';
import { RolesGuard } from 'src/Auth/roles.guard';
import { FiszkaService } from './fiszka.service';
import { Role } from '@prisma/client';
import { Roles } from 'src/Auth/decorators/roles.decorator';
import { HttpExceptionFilter } from 'src/common/validation/http-exception.filter';
import { GetFiszkaDto } from './dto/GetFiszka.dto';
import { EditFiszkaDto } from './dto/EditFiszka.dto';
import { CreateFiszkaDto } from './dto/CreateFiszka.dto';
import { AddgroupDto } from './dto/addGroup.dto';
import { ReqInfo, ReqUserInfo } from 'src/Auth/decorators/reqinfo.decorator';
import { OptionsPipe } from './options.pipe';

@Controller('fiszka')
@UseGuards(RolesGuard)
export class FiszkaController {
  constructor(private fiszkaService: FiszkaService) { }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.RD, Role.CA,Role.USER)
  @Get('findall/:idInterview')
  async findAllByInterview(@Param('idInterview') idInterview: string, @Query('self', OptionsPipe) self: number, @ReqInfo() reqInfo: ReqUserInfo): Promise<GetFiszkaDto[]> {

    if (reqInfo.role.match(Role.USER)) {
      return this.fiszkaService.findAllByUser(idInterview, reqInfo.id)
    } else {
      if (self) {
        return this.fiszkaService.findAllByUser(idInterview, reqInfo.id)
      } else {
        return this.fiszkaService.findAllByInterview(idInterview)
      }
    }
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.RD, Role.CA)
  @Get('findallforuser/:idEmployee')
  async findAllForUser(@Param('idEmployee') userId: string): Promise<GetFiszkaDto[]> {
    return this.fiszkaService.findAllForUser(userId)
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA)
  @Get('findallforregion')
  async findAllForRegion(@ReqInfo() reqInfo: ReqUserInfo): Promise<GetFiszkaDto[]> {
    return this.fiszkaService.findAllForRegion(reqInfo.region)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.RD, Role.CA)
  @Get('findallforgroup/:idGroup')
  async findAllForGroup(@Param('idGroup') idGroup: string,@ReqInfo() reqInfo: ReqUserInfo): Promise<GetFiszkaDto[]> {
    return this.fiszkaService.findAllForGroup(idGroup,reqInfo.region)
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.CA, Role.RD)
  @Patch('editfiszka/:idFiszka')
  async editFiszkaRd(@Param('idFiszka') idFiszka: string, @ReqInfo() reqInfo: ReqUserInfo, @Body() body: EditFiszkaDto): Promise<EditFiszkaDto | ForbiddenException> {
    return this.fiszkaService.editFiszkaRd(idFiszka, reqInfo.id, body)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER)
  @Patch('editfiszka/:idFiszka/:idInterview')
  async editFiszkaUser(@Param('idFiszka') idFiszka: string, @Param('idInterview') idInterview: string, @ReqInfo() reqInfo: ReqUserInfo, @Body() body: EditFiszkaDto): Promise<EditFiszkaDto | ForbiddenException> {
    return this.fiszkaService.editFiszkaUser(idFiszka, reqInfo.id, idInterview, body)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER, Role.MGTMT, Role.CA, Role.RD)
  @Post('addfiszki/:idInterview')
  async addFiszki(@Param('idInterview') idInterview: string, @ReqInfo() reqInfo: ReqUserInfo, @Body() body: CreateFiszkaDto) {
    return this.fiszkaService.addFiszki(idInterview, reqInfo.id, body)
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.MGTMT, Role.CA, Role.RD)
  @Delete('deletefiszka/:idFiszki')
  async deleteFiszka(@Param('idFiszki') idFiszka: string) {
    return this.fiszkaService.deleteFiszka(idFiszka)
  }
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.USER)
  @Delete('deletefiszka/:idFiszki')
  async deleteFiszkaByUser(@Param('idFiszki') idFiszka: string, @ReqInfo() reqInfo: ReqUserInfo) {
    return this.fiszkaService.deleteFiszkaByMy(idFiszka, reqInfo.id)
  }
  //do testu
  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.CA, Role.RD)
  @Patch('addgroup/:idgroup')
  async addGroup(@Param('idgroup') idGroup: string, @Body() body: AddgroupDto) {
    return this.fiszkaService.addGroup(idGroup, body)
  }

  @UseFilters(new HttpExceptionFilter())
  @Roles(Role.CA, Role.RD)
  @Delete('deletefiszka/:idfiszki/fromgroup/:idgroup')
  async deleteFiszkaFromGroup(@Param('idgroup') idGroup: string,@Param('idfiszki') idFiszki: string ):Promise<void> {
    return this.fiszkaService.deleteFiszkaFromGroup(idGroup, idFiszki)
  }
  // wszystkie fiszki z regionu

  // dodanie fiszki do danej grupy

}
