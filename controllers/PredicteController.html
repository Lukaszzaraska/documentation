import { Body, Controller, Get, Param, Req, UseFilters, UseGuards } from '@nestjs/common';
import { Role } from '@prisma/client';
import { Roles } from 'src/Auth/decorators/roles.decorator';
import { HttpExceptionFilter } from 'src/common/validation/http-exception.filter';
import { PredicteService } from './predicte.service';
import { RolesGuard } from 'src/Auth/roles.guard';
import { SendFiszkiDto } from './dto/GetClassFromfiszki.dto';

@Controller('predicte')
@UseGuards(RolesGuard)
export class PredicteController {
    constructor(private predicteService: PredicteService) { }
    // układanie dzieci do konkretnych grup
    // minnimum dzieci w grupie 7 
    // max w grupie 9
    // pokazać wszystkie dni kiedy zgłoszone dzieci (na podstawie klasy) kończą o tej samej godzinie + zmienna czy dzieci mogą czekać np.1 zajęcia na świetlicy
    // czy ilość dzieci w grupach jest aktualna, czy dzieci mogą czekać, czy łączenie dzieci może być między klasami, czy przyjmujemy indywidualne plany zajęć dzieci do uwzględnienia, co jeśli jest sytuacja że dzieci chętnych jest 13 (1 grupa=9 a druga = ?)
    @UseFilters(new HttpExceptionFilter())
    @Roles(Role.RD, Role.CA)
    @Get('group/:IdSchool')
    async Predicte(@Req() req: Request, @Param('IdSchool') IdSchool: string, @Body() body: SendFiszkiDto[]) {
        const region: string = req['user'].region;
        return this.predicteService.Predicte(region, body)

    }
    // Potrzebne dane do pierwszego etapu
    // -Id szkoły
    // -Lista dzieci chętnych 
    // -plany zajęć dzieci które są chętne 
    // Otrzymujemy 
    // Podział 
}
