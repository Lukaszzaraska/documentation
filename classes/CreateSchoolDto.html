import { Status,Type} from "@prisma/client"
import { IsAlpha, IsAlphanumeric, IsAscii, IsEmail, IsEmpty, IsNotEmpty, IsNumber, IsOptional, IsPhoneNumber, IsPostalCode, Length, Max, Min, MinLength } from "class-validator"

export class CreateSchoolDto {
    
    @IsAscii()
    @IsNotEmpty()
    name: string

    @IsAlpha("pl-PL")
    @IsNotEmpty()
    city: string

    @IsNotEmpty()
    status:Status

    @IsNotEmpty()
    type: Type

    @IsNotEmpty()
    cooperation:boolean

    @IsPhoneNumber("PL")
    @IsNotEmpty()
    phone: string

    @IsAlpha("pl-PL")
    @IsNotEmpty()
    province: string

    @IsEmail()
    @IsOptional()
    email: string

    @IsEmpty()
    region: string

    @IsAlpha("pl-PL")
    @IsNotEmpty()
    street: string

    @Length(1,5)
    @IsAlphanumeric("pl-PL")
    @IsNotEmpty()
    suiteNumber:string

    @IsPostalCode("PL")
    @IsNotEmpty()
    postCode: string

    @Min(0)
    @Max(50)
    @IsNumber()
    @IsOptional()
    numberOfClasses: number

    @MinLength(1)
    @IsAscii()
    @IsOptional()
    comments: string
}
