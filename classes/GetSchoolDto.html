import { Province, Status, Type } from "@prisma/client"
import { IsEnum, IsNotEmpty, IsOptional } from "class-validator"

export class GetSchoolDto {
    @IsNotEmpty()
    id: string

    @IsNotEmpty()
    name: string

    @IsNotEmpty()
    region: string

    @IsNotEmpty()
    @IsEnum(Status)
    status: Status

    @IsNotEmpty()
    city: string

    @IsNotEmpty()
    @IsEnum(Type)
    type: Type

    @IsNotEmpty()
    cooperation: boolean

    @IsNotEmpty()
    phone: string

    @IsNotEmpty()
    @IsEnum(Province)
    province: Province

    @IsOptional()
    email: string

    @IsNotEmpty()
    street: string

    @IsNotEmpty()
    suiteNumber: string

    @IsNotEmpty()
    postCode: string

    @IsOptional()
    numberOfClasses: number

    @IsOptional()
    comments: string
}
