import { Injectable } from '@nestjs/common';
import { PrismaService } from 'prisma/prisma.service';
import { CreateRentalCostDto } from './dto/createRentalCost.dto';
import { EditRentalCostDto } from './dto/editRentalCost.dto';
import { GetRentalCostDto } from './dto/getRentalCost.dto';

@Injectable()
export class RentalCostService {
    constructor(private prisma: PrismaService) { }

    async getByIdForMgtmt(idSchool: string): Promise<GetRentalCostDto> {
        return this.prisma.rentalCost.findUniqueOrThrow({
            where: { schoolId: idSchool },
            select: {
                classroomHour: true,
                clockHour: true
            }
        })

    }

    async getByIdForRd(idSchool: string, region: string): Promise<GetRentalCostDto> {
        const Find = this.prisma.school.findUniqueOrThrow({
            where: {
                unique_school_region_id: {
                    id: idSchool,
                    region: region
                }
            },

            select: {
                rentalCost: {
                    select: {
                        id: true,
                        classroomHour: true,
                        clockHour: true
                    }
                }
            }
        })
        return Find.rentalCost()
    }
    async create(idSchool: string, body: CreateRentalCostDto, region: string): Promise<CreateRentalCostDto> {
        return this.prisma.rentalCost.create({
            data: {
                classroomHour: body.classroomHour,
                clockHour: body.clockHour,
                school: {
                    connect: {
                        unique_school_region_id: {
                            id: idSchool,
                            region: region
                        }
                    }
                }
            }

        })
    }
    async patch(idSchool: string, body: EditRentalCostDto, region: string): Promise<EditRentalCostDto> {
        const UpdateInSchool = this.prisma.school.update({
            where: {
                unique_school_region_id: {
                    id: idSchool,
                    region: region
                }
            },
            data: {
                rentalCost: {
                    update: {
                        classroomHour: body.classroomHour,
                        clockHour: body.clockHour
                    }
                }
            }
        })
        return UpdateInSchool.rentalCost()
    }
}
