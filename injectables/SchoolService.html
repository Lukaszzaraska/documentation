import { Injectable } from '@nestjs/common';
import { PrismaService } from 'prisma/prisma.service';
import { EditSchoolDto } from './dto/editSchool.dto';
import { CreateSchoolDto } from './dto/createSchool.dto';
import { GetSchoolDto } from './dto/getSchool.dto';



@Injectable()
export class SchoolService {
    constructor(private prisma: PrismaService) { }


    async findAll(region: string): Promise<GetSchoolDto[]> {

        return this.prisma.school.findMany({
            where: {
                region: region
            },
            select: {
                id: true,
                name: true,
                region: true,
                status: true,
                city: true,
                type: true,
                cooperation: true,
                phone: true,
                province: true,
                email: true,
                street: true,
                suiteNumber: true,
                postCode: true,
                numberOfClasses: true,
                comments: true
            }
        })
    }

    async findById(schoolId: string, region: string): Promise<GetSchoolDto> {

        return this.prisma.school.findUniqueOrThrow({
            where: {
                unique_school_region_id: { region: region, id: schoolId }
            },
            select: {
                id: true,
                name: true,
                region: true,
                status: true,
                city: true,
                type: true,
                cooperation: true,
                phone: true,
                province: true,
                email: true,
                street: true,
                suiteNumber: true,
                postCode: true,
                numberOfClasses: true,
                comments: true
            }
        })

    }

    async findAllForMGTM(): Promise<GetSchoolDto[]> {
        return this.prisma.school.findMany({
            select: {
                id: true,
                name: true,
                region: true,
                status: true,
                city: true,
                type: true,
                cooperation: true,
                phone: true,
                province: true,
                email: true,
                street: true,
                suiteNumber: true,
                postCode: true,
                numberOfClasses: true,
                comments: true
            }
        })
    }

    async findByIdForMGTM(schoolId: string): Promise<GetSchoolDto> {
        return this.prisma.school.findUniqueOrThrow({
            where: {
                id: schoolId
            },
            select: {
                id: true,
                name: true,
                region: true,
                status: true,
                city: true,
                type: true,
                cooperation: true,
                phone: true,
                province: true,
                email: true,
                street: true,
                suiteNumber: true,
                postCode: true,
                numberOfClasses: true,
                comments: true
            }
        })
    }

    async editSchool(schoolId: string, region: string, body: EditSchoolDto): Promise<EditSchoolDto> {
        return this.prisma.school.update({
            where: {
                unique_school_region_id: { region: region, id: schoolId }
            },
            data: body
        })
    }

    async createSchool(region: string, body: CreateSchoolDto):Promise<CreateSchoolDto>{
        let school = this.prisma.school.create({
            data: {
                name: body.name,
                region: region,
                status: body.status,
                city: body.city,
                type: body.type,
                cooperation: body.cooperation,
                phone: body.phone,
                province: body.province,
                email: body.email,
                street: body.street,
                suiteNumber: body.suiteNumber,
                postCode: body.postCode,
                numberOfClasses: body.numberOfClasses,
                comments: body.comments
            }
        })
        return school
    }
}
