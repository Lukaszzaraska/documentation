import { HttpException, HttpStatus, Injectable } from '@nestjs/common';
import { PrismaService } from 'prisma/prisma.service';
import { CreateSheduleDto } from './dto/createShedule.dto';
import { EditSheduleDto } from './dto/editShedule.dto';

@Injectable()
export class SheduleService {
    constructor(private prisma: PrismaService) { }

    async create(body: CreateSheduleDto, IdSchool: string, region: string) {
        return this.prisma.shedule.create({
            data: {
                classDigit: body.classDigit,
                classLetter: body.classLetter,
                date: body.date,
               

                school: {
                    connect: {
                        unique_school_region_id: {
                            id: IdSchool,
                            region: region
                        }
                    }
                }
            }

        })
    }

    async edit(body: EditSheduleDto, IdShedule: string) {
        return this.prisma.shedule.update({
            where: {
               id:IdShedule,
               
            },
            data:{
                classDigit:body.classDigit,
                classLetter:body.classLetter,
                date:body.date,
            }
            }
        )
    }

    async findAll(region: string, IdSchool: string) {
        const shedule = this.prisma.shedule.findMany({
            where: {
                school: {
                    region: region,
                    id: IdSchool
                }
            },
            select: {
                id: true,
                sheduleDays: true,
                classDigit: true,
                classLetter: true,
                date: true
            }
        })
        if((await shedule).length===0){
            throw new HttpException("Not found data with this ID",HttpStatus.NOT_FOUND)
        }
        return shedule
    }
    async findById(region: string, IdShedule: string) {
        return this.prisma.school.findFirst({
            where: {
                region: region,
                shedule: {
                    some: {
                        id: IdShedule
                    }
                }
            },
            select: {
                shedule: {
                    select: {
                        id: true,
                        sheduleDays: true,
                        classDigit: true,
                        classLetter: true,
                        date: true
                    }
                }
            }
        });
    }
}
