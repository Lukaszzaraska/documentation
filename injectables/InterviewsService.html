import { ForbiddenException, Injectable } from '@nestjs/common';
import { PrismaService } from 'prisma/prisma.service';
import { EditInterViewDto } from './dto/editInterView.dto';
import { AddUserInterViewDto } from './dto/addUserInterView.dto';
import { CreateInterViewDto } from './dto/createInterView.dto';
import { RemoveUserInterViewDto } from './dto/removeUserInterview.dto';
import { GetUserInterViewDto } from './dto/getUserInterView.dto';
import { GetInterViewsDto } from './dto/GetInterViews.dto';
import { GetFindByIdDto } from './dto/getFindById.dto';


@Injectable()
export class InterviewsService {
    constructor(private prisma: PrismaService) { }


    async findAllByUser(idUser: string): Promise<GetUserInterViewDto[]> {
        return this.prisma.interviews.findMany({

            where: {
                userInterviews: {
                    some: {
                        userId: idUser
                    }
                }
            },

            select: {
                userInterviews: true,
                id: true,
                classNumber: true,
                termsNumber: true,
                date: true

            }
        })
    }



    async findMyBySchool(IdSchool: string, idUser: string): Promise<GetUserInterViewDto[]> {

        const interviews = this.prisma.interviews.findMany({
            where: {
                school: {
                    id: IdSchool
                },
                userInterviews: {
                    some: {
                        userId: idUser
                    }
                }
            },
            select: {
                id: true,
                classNumber: true,
                termsNumber: true,
                date: true
            }
        });

        return interviews;

    }

    async findById(region: string, IdInterview: string): Promise<GetFindByIdDto> {
        return this.prisma.interviews.findUnique({
            where: {
                id: IdInterview,
                school: {
                    region: region
                }
            },
            include: {
                fiszki: true,
                school: true
            },

        })
    }

    async findAllBySchool(IdSchool: string): Promise<GetUserInterViewDto[]> {
        return this.prisma.interviews.findMany({
            where: {
                school: {
                    id: IdSchool
                }
            },
            select: {
                id: true,
                interviewTotalFiszki: true,
                classNumber: true,
                termsNumber: true,
                date: true,

                fiszki: {
                    select: {
                        id: true,
                        name: true,
                        surname: true,
                        classDigit: true,
                        classLetter: true,
                        phone: true,
                        email: true
                    }
                }
            }
        })
    }
    async findAllByRegion(region: string): Promise<GetInterViewsDto[]> {
        const interviews = this.prisma.interviews.findMany({
            where: {
                school: {
                    region: region
                }
            },
            include: {
                school: {
                    select: {
                        id: true,
                        name: true
                    }
                },
                userInterviews: {
                    select: {
                        user: {
                            select: {
                                id: true,
                                name: true,
                                surname: true
                            }
                        }
                    }
                }
            }
        })


        return interviews
    }
    async findByUser(IdUser: string): Promise<GetUserInterViewDto[]> {
        return this.prisma.interviews.findMany({
            where: {
                userInterviews: {
                    some: {
                        userId: IdUser
                    }
                }
            },
            select: {
                id: true,
                interviewTotalFiszki: true,
                classNumber: true,
                termsNumber: true,
                date: true
            }
        })
    }

    async editInterview(IdInterView: string, Data: EditInterViewDto): Promise<any> {

        const InterView = await this.prisma.interviews.update(
            {
                where: {
                    id: IdInterView
                },

                data: {
                    classNumber: Data.classNumber,
                    termsNumber: Data.termsNumber,
                    date: Data.date,
                    userInterviews: {
                        deleteMany: {}
                    }
                },
                include: {
                    userInterviews: true
                }
            }
        );

        const interviewUserCreateData = Data.userInterviews.map(userData => ({
            userId: userData.id,
            interviewId: IdInterView,
        }));

        const createdInterviewUsers = await this.prisma.interviewUser.createMany({
            data: interviewUserCreateData,
            skipDuplicates: true
        });

        return createdInterviewUsers


    }


    async addUserToInterview(interviewId: string, data: AddUserInterViewDto): Promise<ForbiddenException | GetUserInterViewDto> {

        const interview = await this.prisma.interviews.findUnique({
            where: { id: interviewId },
            include: { userInterviews: true },
        });

        if (!interview) {
            return new ForbiddenException(`Interview(id:${interviewId}) not found`)
        }
        const userExisting = await this.prisma.user.findUniqueOrThrow({
            where: {
                id: data.idUser
            }
        })

        const existingUserIds = interview.userInterviews.map((ui) => ui.userId);
        const isUserIdInArray = existingUserIds.includes(data.idUser);
        if (!isUserIdInArray) {
            return this.prisma.interviews.update({
                where: {
                    id: interviewId
                },
                data: {
                    userInterviews: {
                        create: {
                            userId: data.idUser,

                        }
                    }
                }
            });
        } else {
            return new ForbiddenException(`UserId(id:${data.idUser}) it's already in `)
        }

    }

    async removeUserFromInterview(interviewId: string, data: RemoveUserInterViewDto): Promise<any> {
        const userIds = data.idUserList;

        const interview = await this.prisma.interviews.findUnique({
            where: { id: interviewId },
            include: { userInterviews: true },
        });

        if (!interview) {
            return new ForbiddenException(`Interview(id:${interviewId}) not found`)
        }

        const existingUserIds = interview.userInterviews.map((ui) => ui.userId);
        const userInterviewIds = interview.userInterviews.map((ui) => ui.id);
        const removedUserIds = userIds.filter((id) => existingUserIds.includes(id));
        const removedUserInterviewIds = userInterviewIds.filter((id, index) =>
            existingUserIds.includes(userInterviewIds[index]));

        if (removedUserIds.length === 0) {
            return new ForbiddenException(`One or more Id not found`)

        }

        return this.prisma.interviewUser.deleteMany({
            where: { id: { in: removedUserInterviewIds } },
        });
    }
    async RefreshTotalFiszki() {
        const interviews = await this.prisma.interviews.findMany({
            select: {
                id: true,
                _count: {
                    select: {
                        fiszki: true
                    }
                }
            }
        })
        const updatePromises = interviews.map((interview) => {
            return this.prisma.interviews.update({
                where: { id: interview.id },
                data: { interviewTotalFiszki: interview._count.fiszki }
            });
        });
        await Promise.all(updatePromises);

        return 'Total Fiszki Refreshed';
    }

    async CreateInterview(Data: CreateInterViewDto) {
        const { classNumber, termsNumber, schoolId, users, date } = Data;
        let interviewUsers = [];
        if (users) {
            interviewUsers = users.map(userId => ({
                userId: userId
            }));
        }
        return this.prisma.interviews.create({
            data: {
                classNumber: classNumber,
                termsNumber: termsNumber,
                userInterviews: {
                    create: interviewUsers
                },
                date: date ? date : new Date(), // Nie sprawdzałem czy działa
                school: { connect: { id: schoolId } },
            }
        })
    }
}
