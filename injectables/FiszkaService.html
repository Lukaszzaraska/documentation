import { ForbiddenException, Injectable } from '@nestjs/common';
import { PrismaService } from 'prisma/prisma.service';
import { EditFiszkaDto } from './dto/EditFiszka.dto';
import { CreateFiszkaDto } from './dto/CreateFiszka.dto';
import { AddgroupDto } from './dto/addGroup.dto';
import { GetFiszkaDto } from './dto/GetFiszka.dto';


@Injectable()
export class FiszkaService {
    constructor(private prisma: PrismaService) { }

    async findAllByInterview(idInterview: string): Promise<GetFiszkaDto[]> {
        const fiszka = this.prisma.fiszka.findMany({
            where: {
                interviews: {
                    id: idInterview
                }
            },
            select: {
                id: true,
                name: true,
                surname: true,
                classLetter: true,
                classDigit: true,
                comments: true,
                phone: true,
                email: true,
                resignation: true,
                phoneCalls: true,
                group: true,
                user: {
                    select: {
                        name: true,
                        surname: true
                    }
                }
            }
        });
        return fiszka;
    }
    async findAllByUser(idInterview: string, UserID: string): Promise<GetFiszkaDto[]> {
        return this.prisma.fiszka.findMany({
            where: {
                interviews: {
                    id: idInterview
                },
                user: {
                    id: UserID
                }
            },
            select: {
                id: true,
                name: true,
                surname: true,
                classLetter: true,
                classDigit: true,
                comments: true,
                phone: true,
                email: true,
                resignation: true,
                phoneCalls: true,
                group: true
            }
        });

    }
    async findAllForUser(idEmployee: string): Promise<GetFiszkaDto[]> {
        return this.prisma.fiszka.findMany({
            where: {
                user: {
                    id: idEmployee
                }
            },
            select: {
                id: true,
                name: true,
                surname: true,
                classLetter: true,
                classDigit: true,
                comments: true,
                phone: true,
                email: true,
                resignation: true,
                phoneCalls: true,
                group: true
            }
        });
    }
    async findAllForRegion(region: string): Promise<GetFiszkaDto[]> {
        return this.prisma.fiszka.findMany({
            where: {
                user: {
                    region: region
                }
            },
            select: {
                id: true,
                name: true,
                surname: true,
                classLetter: true,
                classDigit: true,
                comments: true,
                phone: true,
                email: true,
                resignation: true,
                phoneCalls: true,
                group: true,
                user: {
                    select: {
                        name: true,
                        surname: true
                    }
                }
            }
        });
    }
    async findAllForGroup(idGroup: string, region: string): Promise<GetFiszkaDto[]> {
        return this.prisma.fiszka.findMany({
            where: {
                user: {
                    region: region
                },
                group: {
                    id: idGroup
                }
            },
            select: {
                id: true,
                name: true,
                surname: true,
                classLetter: true,
                classDigit: true,
                comments: true,
                phone: true,
                email: true,
                resignation: true,
                phoneCalls: true,
                group: true,
                user: {
                    select: {
                        name: true,
                        surname: true
                    }
                }
            }
        });
    }

    async editFiszkaRd(idFiszka: string, UserID: string, body: EditFiszkaDto): Promise<EditFiszkaDto | ForbiddenException> {
        const fiszka = await this.prisma.fiszka.findUniqueOrThrow({
            where: {
                unique_fiszka_user_id: {
                    id: idFiszka,
                    userId: UserID
                }
            },
        });
        if (!fiszka) {
            return new ForbiddenException(`This fiszka its not yours!`)
        }
        return this.prisma.fiszka.update({
            where: {
                id: idFiszka
            },
            data: body
        })
    }
    async editFiszkaUser(idFiszka: string, UserID: string, idInterview: string, body: EditFiszkaDto): Promise<EditFiszkaDto | ForbiddenException> {
        const fiszka = await this.prisma.fiszka.findUniqueOrThrow({
            where: {
                unique_fiszka_user_id: {
                    id: idFiszka,
                    userId: UserID
                },
                interviewsId: idInterview
            },
        });

        if (!fiszka) {
            return new ForbiddenException(`This fiszka its not yours!`)
        }
        return this.prisma.fiszka.update({
            where: {
                id: idFiszka
            },
            data: body
        })
    }
    async addFiszki(idInterview: string, userId: string, body: CreateFiszkaDto) {

        const createdFiszka = await this.prisma.fiszka.create({
            data: {
                name: body.name,
                surname: body.surname,
                classLetter: body.classLetter,
                classDigit: body.classDigit,
                comments: body.comments,
                phone: body.phone,
                email: body.email,
                userId: userId,
                interviewsId: idInterview
            },
            select: {
                id: true,
                name: true,
                surname: true,
                interviewsId: true,
                user: {
                    select: {
                        name: true,
                        surname: true
                    }
                }
            }
        });

        await this.prisma.interviews.update({
            where: { id: idInterview },
            data: {
                interviewTotalFiszki: {
                    increment: 1
                }
            }
        })

        return createdFiszka;
    }
    async deleteFiszka(IdFiszki: string) {
        return await this.prisma.fiszka.delete({
            where: {
                id: IdFiszki
            },
        })

    }
    async deleteFiszkaByMy(IdFiszki: string, IdUser: string) {
        return await this.prisma.fiszka.delete({
            where: {
                id: IdFiszki,
                userId: IdUser
            },
        })

    }
    async addGroup(IdGroup: string, body: AddgroupDto) {
        const updatedGroup = await this.prisma.group.update({
            where: {
                id: IdGroup
            },
            data: {
                fiszki: {
                    connect: {
                        id: body.fiszka.id
                    }
                }
            },
            include: {
                fiszki: true
            }
        });

        return updatedGroup;
    }

    async deleteFiszkaFromGroup(idGroup: string, idFiszki: string): Promise<void> {
        const deleteFiszkaFromGroup = await this.prisma.group.update({
            where: {
                id: idGroup
            },
            data: {
                fiszki: {
                    disconnect: {
                        id: idFiszki
                    }
                }
            }

        })
    }
}
